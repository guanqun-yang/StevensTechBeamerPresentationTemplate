% colorblock: \begin{colorblock}[text color]{background color}{title}
\newenvironment{colorblock}[3][black]{%
  \begingroup
  \setbeamercolor{block title}{fg=#1, bg=#2}
  \setbeamercolor{block body}{fg=#1, bg=#2}
  \begin{block}{#3}
}{%
  \end{block}
  \endgroup
}

}{%

\newcounter{outlineidx}

\newcommand{\outlineitem}[3]{% #1=section num, #2=nickname, #3=full title
  \setcounter{outlineidx}{#1}%
  \ifnum\value{section}=\value{outlineidx}%
    \par{\large\textbf{\textcolor{stevensred}{#2}}}\par\vspace{-0.2em}%
    {\small\textcolor{stevensred}{#3}}\par\vspace{0.5em}%
  \else%
    \par{\large\textcolor{stevensgray!70}{#2}}\par\vspace{-0.2em}%
    {\small\textcolor{stevensgray!50}{#3}}\par\vspace{0.5em}%
  \fi%
}

    \begin{minipage}{0.15\textwidth}
      \raggedleft
      \includegraphics[height=1.4cm]{assets/logo_RGB}
    \end{minipage}
    \par\vspace{0.5em}
    {\color{stevensred}\rule{\textwidth}{1pt}}
    \par\vspace{0.6em}
    \outlineitem{1}{\secnameI}{\secfullI}%
    \outlineitem{2}{\secnameII}{\secfullII}%
    \outlineitem{3}{\secnameIII}{\secfullIII}%
    \outlineitem{4}{\secnameIV}{\secfullIV}%
  \end{frame}
}
